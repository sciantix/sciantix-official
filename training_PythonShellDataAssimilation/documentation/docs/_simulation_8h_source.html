296  void InterGranularBubbleBehaviour()
297  {
298  const double pi = CONSTANT_NUMBERS_H::MathConstants::pi;
299 
300  // Vacancy concentration
301  sciantix_variable[sv["Intergranular vacancies per bubble"]].setFinalValue(
302  solver.LimitedGrowth(sciantix_variable[sv["Intergranular vacancies per bubble"]].getInitialValue(),
303  model[sm["Intergranular bubble evolution"]].getParameter(),
304  physics_variable[pv["Time step"]].getFinalValue()
305  )
306  );
307 
308  // Grain-boundary bubble volume
309  double vol(0);
310  for (std::vector<System>::size_type i = 0; i != sciantix_system.size(); ++i)
311  // This for loop slides over only the stable gases, which determine the grain-boundary bubble dynamics
312  {
313  if (gas[ga[sciantix_system[i].getGasName()]].getDecayRate() == 0.0)
314  {
315  vol += sciantix_variable[sv["Intergranular " + sciantix_system[i].getGasName() + " atoms per bubble"]].getFinalValue() *
316  gas[ga[sciantix_system[i].getGasName()]].getVanDerWaalsVolume();
317  }
318  }
319  vol += sciantix_variable[sv["Intergranular vacancies per bubble"]].getFinalValue() * matrix[0].getSchottkyVolume();
320  sciantix_variable[sv["Intergranular bubble volume"]].setFinalValue(vol);
321 
322  // Grain-boundary bubble radius
323  sciantix_variable[sv["Intergranular bubble radius"]].setFinalValue(
324  0.620350491 * pow(sciantix_variable[sv["Intergranular bubble volume"]].getFinalValue() / (matrix[0].getLenticularShapeFactor()), 1. / 3.));
325 
326  // Grain-boundary bubble area
327  sciantix_variable[sv["Intergranular bubble area"]].setFinalValue(
328  pi * pow(sciantix_variable[sv["Intergranular bubble radius"]].getFinalValue() * sin(matrix[0].getSemidihedralAngle()), 2));
329 
330  // Grain-boundary bubble coalescence
331  double dbubble_area = sciantix_variable[sv["Intergranular bubble area"]].getIncrement();
332  sciantix_variable[sv["Intergranular bubble concentration"]].setFinalValue(
333  solver.BinaryInteraction(sciantix_variable[sv["Intergranular bubble concentration"]].getInitialValue(), 2.0, dbubble_area));
334 
335  // Conservation
336  for (std::vector<System>::size_type i = 0; i != sciantix_system.size(); ++i)
337  {
338  if (gas[ga[sciantix_system[i].getGasName()]].getDecayRate() == 0.0)
339  {
340  sciantix_variable[sv["Intergranular " + sciantix_system[i].getGasName() + " atoms per bubble"]].rescaleFinalValue(
341  sciantix_variable[sv["Intergranular bubble concentration"]].getInitialValue() / sciantix_variable[sv["Intergranular bubble concentration"]].getFinalValue()
342  );
343  }
344  }
345 
346  double n_at(0);
347  for (std::vector<System>::size_type i = 0; i != sciantix_system.size(); ++i)
348  {
349  if (gas[ga[sciantix_system[i].getGasName()]].getDecayRate() == 0.0)
350  n_at += sciantix_variable[sv["Intergranular " + sciantix_system[i].getGasName() + " atoms per bubble"]].getFinalValue();
351  }
352  sciantix_variable[sv["Intergranular atoms per bubble"]].setFinalValue(n_at);
353 
354  sciantix_variable[sv["Intergranular vacancies per bubble"]].rescaleFinalValue(
355  sciantix_variable[sv["Intergranular bubble concentration"]].getInitialValue() / sciantix_variable[sv["Intergranular bubble concentration"]].getFinalValue()
356  );
357 
358  vol = 0.0;
359  for (std::vector<System>::size_type i = 0; i != sciantix_system.size(); ++i)
360  {
361  if (gas[ga[sciantix_system[i].getGasName()]].getDecayRate() == 0.0)
362  {
363  vol += sciantix_variable[sv["Intergranular " + sciantix_system[i].getGasName() + " atoms per bubble"]].getFinalValue() *
364  gas[ga[sciantix_system[i].getGasName()]].getVanDerWaalsVolume();
365  }
366  }
367  vol += sciantix_variable[sv["Intergranular vacancies per bubble"]].getFinalValue() * matrix[0].getSchottkyVolume();
368  sciantix_variable[sv["Intergranular bubble volume"]].setFinalValue(vol);
369 
370  sciantix_variable[sv["Intergranular bubble radius"]].setFinalValue(
371  0.620350491 * pow(sciantix_variable[sv["Intergranular bubble volume"]].getFinalValue() / (matrix[0].getLenticularShapeFactor()), 1. / 3.));
372 
373  // sciantix_variable[sv["Intergranular bubble area"]].setInitialValue(sciantix_variable[sv["Intergranular bubble area"]].getFinalValue());
374  sciantix_variable[sv["Intergranular bubble area"]].setFinalValue(
375  pi * pow(sciantix_variable[sv["Intergranular bubble radius"]].getFinalValue() * sin(matrix[0].getSemidihedralAngle()), 2));

383  // F0

513 
514  // ODE for the intergranular fractional intactness: this equation accounts for the reduction of the intergranular fractional intactness following a temperature transient
515  // df / dT = - dm/dT f
516  sciantix_variable[sv["Intergranular fractional intactness"]].setFinalValue(
517  solver.Decay(sciantix_variable[sv["Intergranular fractional intactness"]].getInitialValue(),
518  model[sm["Grain-boundary micro-cracking"]].getParameter().at(0), // 1st parameter = microcracking parameter
519  0.0,
520  history_variable[hv["Temperature"]].getIncrement()
521  )
522  );
523 
524  // ODE for the intergranular fractional coverage: this equation accounts for the reduction of the intergranular fractional coverage following a temperature transient
525  // dFc / dT = - ( dm/dT f) Fc
526  sciantix_variable[sv["Intergranular fractional coverage"]].setFinalValue(
527  solver.Decay(sciantix_variable[sv["Intergranular fractional coverage"]].getInitialValue(),
528  model[sm["Grain-boundary micro-cracking"]].getParameter().at(0) * sciantix_variable[sv["Intergranular fractional intactness"]].getFinalValue(),
529  0.0,
530  history_variable[hv["Temperature"]].getIncrement()
531  )
532  );
533 
534  // ODE for the saturation fractional coverage: this equation accounts for the reduction of the intergranular saturation fractional coverage following a temperature transient
535  // dFcsat / dT = - (dm/dT f) Fcsat
536  sciantix_variable[sv["Intergranular saturation fractional coverage"]].setFinalValue(

542  )
543  );

553  )
554  );

582  sciantix_variable[sv["Intergranular bubble volume"]].rescaleInitialValue(pow(similarity_ratio, 1.5));
583  sciantix_variable[sv["Intergranular bubble radius"]].rescaleInitialValue(pow(similarity_ratio, 0.5));
584  sciantix_variable[sv["Intergranular vacancies per bubble"]].rescaleInitialValue(pow(similarity_ratio, 1.5));

586  for (std::vector<System>::size_type i = 0; i != sciantix_system.size(); ++i)

590  }

592  double n_at(0);

617  void GrainBoundaryVenting()
618  {
619  if (!int(input_variable[iv["iGrainBoundaryVenting"]].getValue())) return;
620 
621  double sigmoid_variable;
622  sigmoid_variable = sciantix_variable[sv["Intergranular fractional coverage"]].getInitialValue() *
623  exp(-sciantix_variable[sv["Intergranular fractional intactness"]].getIncrement());
624 
625  // Vented fraction
626  sciantix_variable[sv["Intergranular vented fraction"]].setFinalValue(
627  1.0 /
628  pow((1.0 + model[sm["Grain-boundary venting"]].getParameter().at(0) *
629  exp(-model[sm["Grain-boundary venting"]].getParameter().at(1) *
630  (sigmoid_variable - model[sm["Grain-boundary venting"]].getParameter().at(2)))),
631  (1.0 / model[sm["Grain-boundary venting"]].getParameter().at(0)))
632  );
633 
634  // Venting probability
635  sciantix_variable[sv["Intergranular venting probability"]].setFinalValue(
636  (1.0 - sciantix_variable[sv["Intergranular fractional intactness"]].getFinalValue())
637  + sciantix_variable[sv["Intergranular fractional intactness"]].getFinalValue() * sciantix_variable[sv["Intergranular vented fraction"]].getFinalValue()
638  );
639 
640  // Gas is vented by subtracting a fraction of the gas concentration at grain boundaries arrived from diffusion

648  sciantix_variable[sv[sciantix_system[i].getGasName() + " at grain boundary"]].getIncrement()
649  )
